#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main() {
    char name[50];
    int choice, score = 0;
    int playAgain = 1;

    srand(time(0)); // random generator

    printf("Enter your name: ");
    scanf("%s", name);

    while (playAgain == 1) {
        score = 0;

        printf("\n🌲 Welcome %s to the Adventure Game 🌲\n", name);
        printf("*footsteps*\nYou are in a dark forest...\n");

        printf("1. Go left\n2. Go right\n");
        scanf("%d", &choice);

        if (choice == 1) {
            printf("\n*roar* A wild animal appears!\n");

            int event = rand() % 2; // random event

            if (event == 0) {
                printf("1. Fight\n2. Run\n");
                scanf("%d", &choice);

                if (choice == 1) {
                    printf("\nYou defeated the animal! 🏆\n");
                    score += 50;
                } else {
                    printf("\nYou escaped safely 😅\n");
                    score += 20;
                }
            } else {
                printf("\nThe animal attacked suddenly! 💀\n");
                score -= 30;
            }

        } else if (choice == 2) {
            printf("\nYou found a treasure chest! 🎁\n");
            score += 40;

            int luck = rand() % 2;

            if (luck == 0) {
                printf("*click* It was a trap! 💀\n");
                score -= 50;
            } else {
                printf("You got gold coins! 💰\n");
                score += 30;
            }

        } else {
            printf("\nInvalid choice!\n");
        }

        printf("\n🎯 Your Score: %d\n", score);

        printf("\nDo you want to play again?\n1. Yes\n2. No\n");
        scanf("%d", &playAgain);
    }

    printf("\nThanks for playing, %s! 👋\n", name);

    return 0;
}
